এই ফোল্ডারে নিচের লজিক অনুযায়ী কর। 

z_insta_post_db  instagram_post\gopals_diary\index.html ও instagram_post\kobitar_sohor\index.html এ ব্যবহার হবে। 

আর  z_insta_sche_gopals_diary ও z_insta_sche_kobitar_sohor আলাদা আলাদা ভাবে ব্যবহার instagram_post\gopals_diary\index.ts

ও 
instagram_post\kobitar_sohor\index.ts
instagram_post\kobitar_sohor\schiduler.html 



-------------  z_insta_post_db  এর status_kobitar_sohor column টি status_kobitar_sohor table  এ status আপডেট করবে। (এখন status লিখা কোডে আছে) যদি index.html থেকে ম্যানুয়্যালি অথবা index.ts এর মাধ্যমে supabase edge function থেকে পোস্ট হয়। 



create table public.z_insta_post_db (
  id uuid not null default gen_random_uuid (),
  created_at timestamp with time zone not null default timezone ('utc'::text, now()),
  image_url text not null,
  caption text null,
  platform text null default 'instagram'::text,
  status_kobitar_sohor text null default 'pending'::text,
  published_at_kobitar_sohor timestamp with time zone null,
  user_id uuid null,
  post_section text null,
  post_sl bigint generated by default as identity not null,
  status_gopals_diary text null,
  published_at_gopals_diary timestamp with time zone null,
  constraint scheduled_posts_pkey primary key (id),
  constraint scheduled_posts_user_id_fkey foreign KEY (user_id) references auth.users (id)
) TABLESPACE pg_default;

create table public.z_insta_sche_gopals_diary (
  id uuid not null default gen_random_uuid (),
  created_at timestamp with time zone null default now(),
  image_url text not null,
  caption text null,
  platform text null default 'instagram'::text,
  status text null default 'pending'::text,
  schedule_at timestamp with time zone null default now(),
  published_at timestamp with time zone null,
  post_section text null default 'other'::text,
  post_sl bigint generated by default as identity not null,
  error_log text null,
  user_id uuid null,
  constraint instagram_posts_gopals_diary_pkey primary key (id),
  constraint instagram_posts_gopals_diary_user_id_fkey foreign KEY (user_id) references auth.users (id)
) TABLESPACE pg_default;

create index IF not exists instagram_posts_gopals_diary_status_schedule_at_idx on public.z_insta_sche_gopals_diary using btree (status, schedule_at) TABLESPACE pg_default
where
  (status = 'pending'::text);
