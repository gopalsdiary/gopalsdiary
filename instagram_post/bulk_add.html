create table public.z_insta_post_db ( id uuid not null default gen_random_uuid (), created_at timestamp with time zone not null default timezone ('utc'::text, now()), image_url text not null, caption text null, platform text null default 'instagram'::text, status_kobitar_sohor text null default 'pending'::text, published_at_kobitar_sohor timestamp with time zone null, user_id uuid null, post_section text null, post_sl bigint generated by default as identity not null, status_gopals_diary text null, published_at_gopals_diary timestamp with time zone null, constraint scheduled_posts_pkey primary key (id), constraint scheduled_posts_user_id_fkey foreign KEY (user_id) references auth.users (id) ) TABLESPACE pg_default; এই টেবিলে image_url (সুপবাবেস বাকেট এর "insta_post_bucket" ফোল্ডারে সেভ হবে ) caption ("📋 Custom 1" from instagram_post\kobitar_sohor\index.html) post_section (ড্রপ ডাউন হবে) Files > Buckets > insta_post_bucket অনেকগুলো ছবি একসাথে ইনপোর্ট করা যাবে। ছবিগুলো ৪৭০kb এর বেশি হলে, কমপ্রেস করে 470kb এর মধ্যে কোয়ালিটি নষ্ট না করে কমপ্রেস করে আপলোড করবে। "📋 Custom 1" সিলেক্ট করলে, সব গুলোতে এপ্লাই হবে, এবং কাস্টম ক্যাপশন লিখা যাবে। পোস্ট সেকশন ও প্রথমে যেটি সিলেক্ট করবো, সেটি সব গুলোর জন্য এপ্লাই হবে, এরপর ম্যানুয়্যালি চেঞ্জ করা যাবে। নিচে সেভ বাটন দিলে তারপর সুপাবেস টেবিলে সেভ করবে। এর আগে কমপ্রেস করে সাথে সাথে বাকেট এ সেভ করে লিঙ্ক দেখাবে।
